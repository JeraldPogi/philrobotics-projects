34  void setupSerial(unsigned int iBaudrate)
35  {
36  //UART_REGISTER |= UART_RX_MASK; UART_REGISTER &= ~UART_TX_MASK;
37 
38  TXFiFo.iIn = TXFiFo.iOut = RXFiFo.iIn = RXFiFo.iOut = 0;
39 
40  BIT_TXSTA_TX9 = 0; //8-bit
41  BIT_TXSTA_TXEN = 1; //Enable Transmit
42  BIT_TXSTA_SYNC = 0; //Asyncronous Mode
43  //BIT_TXSTA_TRMT = 1; //TSR is empty
44 
45  BIT_TXSTA_BRGH = 1; //High Speed Asyncronous // SPBRG = FOSC/(16*baud) - 1
46  if(iBaudrate < 4883) // @20MHz clk
47  BIT_TXSTA_BRGH = 0; // Low Speed Asyncronous // SPBRG = FOSC/(64*baud) - 1
48 
49  switch(iBaudrate){ // FOSC = 20MHz
50  case 1200:
51  REGISTER_SPBRG = 255 /*259*/; // 1220.7
52  break;
53  case 4800:
54  case 19200:
55  REGISTER_SPBRG = 64; // 4807.7 or 19230.8
56  break;
57  case 38400:
58  REGISTER_SPBRG = 32; // 39062.5
59  break;
60  case 57600:
61  REGISTER_SPBRG = 21; // 56818.2
62  break;
63  case 115200:
64  REGISTER_SPBRG = 10; // 113636.4
65  break;
66  case 230400:
67  REGISTER_SPBRG = 4; // 250000.0 (8.5% error!)
68  break;
69  case 2400:
70  case 9600:
71  default: // baud defaults
72  REGISTER_SPBRG = 129; // 2403.8 or 9615.4
73  break;
74  }
75 
76  BIT_RCSTA_SPEN = 1; //Enable Serial Port
77  BIT_RCSTA_RX9 = 0; //8-bit
78  BIT_RCSTA_CREN = 1; //Enable Continuous receive

165  void serialRxInterruptHandler(void)
166  {
167  int iTempIn;
168 
169  if (BIT_PIR1_RCIF){
170 
171  RXFiFo.ucBuffer[RXFiFo.iIn] = REGISTER_RCREG;
172  iTempIn = ((RXFiFo.iIn+1) & (BUFFER_SIZE-1));

174  if (iTempIn != RXFiFo.iOut)
