65 //***********************************************************************************
66 // intSource: Interrupt Pin Assignment
67 // INT0 - D8: External Interrupt (RB0)
68 // INT1 - D12: Interrupt on Pin Change (RB4)
69 // INT2 - D13: Interrupt on Pin Change (RB5)
70 // INT3 - PGC: Interrupt on Pin Change (RB6)
71 // INT4 - PGD: Interrupt on Pin Change (RB7)
72 //***********************************************************************************
73 // function: name of the function ISR to call when interrupt occured,
74 // this function must take no parameters and does not return anything
75 //***********************************************************************************
76 // mode: Defines how the interrupt should be triggered
77 // LOW - Trigger if Int Pin is Low (Currently Not Supported)
78 // CHANGE - Trigger if there is a change on interrupt pin state
79 // RISING - Trigger on rising edge
80 // FALLING - Trigger on falling edge
81 //***********************************************************************************
82 void setupUserInterrupt(enum etInterruptSources eIntSource, void(*function)(), enum etInterruptModes eIntMode)
83 {
84  /* External Interrupt */
85  if(INT0 == eIntSource)
86  {
87  #if(EXTINTENABLED == TRUE)
88  /* Set Int Pin to Input */
89  setPinInput(D8);
90 
91  if(RISING == eIntMode)
92  {
93  INTEDG = 1; // Start detecting at rising edge
94  }
95  else if(FALLING == eIntMode)
96  {
97  INTEDG = 0; // Start detecting at falling edge
98  }
99  else if(CHANGE == eIntMode)
100  {
101  /* Set the Inverse of Current Pin State*/
102  if(isPinHigh(D8))
103  {
104  INTEDG = 0;
105  }
106  else
107  {
108  INTEDG = 1;
109  }
110  }
111  else
112  {
113  /* Do Nothing */
114  }
115 
116  mode0 = eIntMode;
117  pt2INT0 = function; // assign INT0 Function Pointer
118  BIT_INTCON_INTF = 0;
119  BIT_INTCON_INTE = 1;
120  #endif
121  }
122  /* Interrupt on PortB Change */
123  else
124  {
125  #if(RBINTENABLED == TRUE)
126  PORTB_DIR = 0x00;
127 
128  /* RB4 Interrupt on Change */
129  if(INT1 == eIntSource)
130  {
131  /* Set Int Pin to Input */
132  setPinInput(D12);
133  PORTB_DIR |= RB4_MASK;
134  mode1 = eIntMode;
135  pt2INT1 = function; // assign INT1 Function Pointer
136  }
137  /* RB5 Interrupt on Change */
138  else if(INT2 == eIntSource)
139  {
140  /* Set Int Pin to Input */
141  setPinInput(D13);
142  PORTB_DIR |= RB5_MASK;
143  mode2 = eIntMode;
144  pt2INT2 = function; // assign INT2 Function Pointer
145  }
146  /* RB6 Interrupt on Change */
147  else if(INT3 == eIntSource)
148  {
149  /* Set Int Pin to Input */
150  REGISTER_TRISB |= RB6_MASK;
151  PORTB_DIR |= RB6_MASK;
152  mode3 = eIntMode;
153  pt2INT3 = function; // assign INT3 Function Pointer
154  }
155  /* RB7 Interrupt on Change */
156  else if(INT4 == eIntSource)
157  {
158  /* Set Int Pin to Input */
159  REGISTER_TRISB |= RB7_MASK;
160  PORTB_DIR |= RB7_MASK;
161  mode4 = eIntMode;
162  pt2INT4 = function; // assign INT4 Function Pointer
163  }
164  else
165  {
166  /* Do Nothing */
167  }
168 
169  /* PORTB States Reference */
170  PORTB_BUFFER = REGISTER_PORTB & PORTB_DIR; // Ignore Outputs
171 
172  BIT_INTCON_RBIF = 0;
173  BIT_INTCON_RBIE = 1;
174  #endif
175  }
176 }

178 void userInterruptHandler(void)
179 {
180  /* Check External Interrupt */
181  if(BIT_INTCON_INTF&&BIT_INTCON_INTE)
182  {
183  BIT_INTCON_INTE = 0; // Temporary Disable Interrupt
184 #if(EXTINTENABLED == TRUE)
185  extInterruptHandler();
186 #endif
187  BIT_INTCON_INTF = 0;
188  BIT_INTCON_INTE = 1; // Reenable Interrupt
189  }
190 
191  /* Check Interrupt on PortB Change */
192  if(BIT_INTCON_RBIF&&BIT_INTCON_RBIE)
193  {
194  BIT_INTCON_RBIE = 0; // Temporary Disable Interrupt
195 #if(RBINTENABLED == TRUE)
196  rbInterruptHandler();
197 #endif
198  BIT_INTCON_RBIF = 0;
199  BIT_INTCON_RBIE = 1; // Reenable Interrupt
200  }
201 }

222 #if(RBINTENABLED == TRUE)
223 static void rbInterruptHandler(void)
224 {
225  int PORTCHANGE_BUFFER;
226 
227  /* Capture Pin Changes */
228  PORTCHANGE_BUFFER = PORTB_BUFFER ^ (REGISTER_PORTB & PORTB_DIR); // Ignore Outputs
229 
230  if(RB4_MASK == (PORTCHANGE_BUFFER&RB4_MASK))
231  {
232  if((RISING == mode1) && isPinLow(D12))
233  {
234  /* Ignore and wait for the next interrupt */
235  }
236  else if((FALLING == mode1) && isPinHigh(D12))
237  {
238  /* Ignore and wait for the next interrupt */
239  }
240  else // CHANGE == mode1 or coincide of correct mode and pin state
241  {
242  /* Call User Function */
243  pt2INT1();
244  }
245  PORTB_BUFFER &= ~RB4_MASK; // clear buffered RB4
246  PORTB_BUFFER |= REGISTER_PORTB & RB4_MASK; // store new RB4
247  }
248 
249  if(RB5_MASK == (PORTCHANGE_BUFFER&RB5_MASK))
250  {
251  if((RISING == mode2) && isPinLow(D13))
252  {
253  /* Ignore and wait for the next interrupt */
254  }
255  else if((FALLING == mode3) && isPinHigh(D13))
256  {
257  /* Ignore and wait for the next interrupt */
258  }
259  else // CHANGE == mode1 or coincide of correct mode and pin state
260  {
261  /* Call User Function */
262  pt2INT2();
263  }
264  PORTB_BUFFER &= ~RB5_MASK; // clear buffered RB5
265  PORTB_BUFFER |= REGISTER_PORTB & RB5_MASK; // store new RB5
266  }
267 
268  if(RB6_MASK == (PORTCHANGE_BUFFER&RB6_MASK))
269  {
270  if((RISING == mode3) && !(REGISTER_PORTB & RB6_MASK))
271  {
272  /* Ignore and wait for the next interrupt */
273  }
274  else if((FALLING == mode3) && (REGISTER_PORTB & RB6_MASK))
275  {
276  /* Ignore and wait for the next interrupt */
277  }
278  else // CHANGE == mode1 or coincide of correct mode and pin state
279  {
280  /* Call User Function */
281  pt2INT3();
282  }
283  PORTB_BUFFER &= ~RB6_MASK; // clear buffered RB6
284  PORTB_BUFFER |= REGISTER_PORTB & RB6_MASK; // store new RB6
285  }
286 
287  if(RB7_MASK == (PORTCHANGE_BUFFER&RB7_MASK))
288  {
289  if((RISING == mode4) && !(REGISTER_PORTB & RB7_MASK))
290  {
291  /* Ignore and wait for the next interrupt */
292  }
293  else if((FALLING == mode4) && (REGISTER_PORTB & RB7_MASK))
294  {
295  /* Ignore and wait for the next interrupt */
296  }
297  else // CHANGE == mode1 or coincide of correct mode and pin state
298  {
299  /* Call User Function */
300  pt2INT4();
301  }
302  PORTB_BUFFER &= ~RB7_MASK; // clear buffered RB7
303  PORTB_BUFFER |= REGISTER_PORTB & RB7_MASK; // store new RB7
304  }
305 
306  PORTB_BUFFER &= PORTB_DIR; // Ensure only inputs are stored
307 }
308 #endif
