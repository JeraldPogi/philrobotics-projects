33 //***********************************************************************************
34 // @20Mhz -> TOSC = 50nS
35 // PWM Period = [(PR2) + 1] * 4 * TOSC * Prescaler
36 // Period(uS) = [PR2+1] * 200nS * Prescaler
37 // 1.22kHz - 200kHz
38 // Prescaler 1
39 // 19.53Khz -> 51.20uS
40 // 200Khz -> 5.00uS
41 //
42 // Prescaler 4
43 // 4.88Khz -> 204.92uS
44 // 19.53Khz -> 51.20uS
45 //
46 // Prescaler 16
47 // 1.22Khz -> 819.67uS
48 // 4.88kHz -> 204.92uS
49 //
50 // ui16Frequency: in 10Hz/Bit Resolution (e.g. 1kHz/10Hz = 100)
51 //***********************************************************************************
52 void setPWMFrequency(uint16_t ui16Frequency)
53 {
54  uint16_t ui16TempVar;
55  uint24_t ui24Period; // 0.01uS/Bit Resolution
56 
57  /* Check Frequency Range */
58  if(ui16Frequency > 20000) // > 200kHz
59  {
60  ui16Frequency = 20000;
61  }
62  else if(ui16Frequency < 122) // < 1.22kHz
63  {
64  ui16Frequency = 122;
65  }
66  else
67  {
68  /* Frequency within Range */
69  /* Do Nothing */
70  }
71 
72  /* Check Prescaler Range */
73  if(ui16Frequency >= 1953) // 19.53kHz - 200kHz
74  {
75  mui8PreScalerVal = 0; // div by 1
76  }
77  else if(ui16Frequency >= 488) // 4.88kHz - 19.53kHz
78  {
79  mui8PreScalerVal = 1; // div by 4
80  }
81  else // 1.22kHz - 4.88kHz
82  {
83  mui8PreScalerVal = 2; // div by 16
84  }
85 
86  mui8PreScaler = mui8PreScalerVal << 1;
87  mui8PreScaler = (uint8_t)1 << mui8PreScaler;// secret :p
88 
89  ui24Period = (uint24_t)(10000000UL / ui16Frequency);
90 
91  ui16TempVar = (uint16_t)20 * mui8PreScaler;
92  ui24Period = ui24Period/ui16TempVar;
93 
94  /* Check Saturation */
95  if(ui24Period > 255)
96  {
97  mui8PR2plus1 = 255;
98  }
99  else
100  {
101  mui8PR2plus1 = (uint8_t)ui24Period;
102  }
103 
104  mc_setPWMPeriod(mui8PR2plus1);
105 }

115 void setPWMDuty(enum ePWMModules ePWM_Module, uint16_t ui16DutyCycle)
116 {
117  uint16_t ui16TempVar;
118 
119  /* Maximum of 100% Only */
120  if(ui16DutyCycle > 1000)
121  {
122  ui16DutyCycle = 1000;
123  }
124 
125  ui16TempVar = (uint16_t)mui8PR2plus1 << 2; // secret :p
126  ui16TempVar = (uint16_t)(((uint32_t)ui16TempVar * ui16DutyCycle) / 1000);
127 
128  if(PWM0 == ePWM_Module)
129  {
130  mc_setPWM0Ton(ui16TempVar);
131  }
132  else if(PWM1 == ePWM_Module)
133  {
134  mc_setPWM1Ton(ui16TempVar);
135  }
136  else
137  {
138  /* Invalid PWM Module */
139  /* Do Nothing */
140  }
141 }
