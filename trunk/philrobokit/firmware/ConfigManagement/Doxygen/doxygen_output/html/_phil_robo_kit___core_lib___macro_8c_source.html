46 void configPin(unsigned char ucPinName, char bDirection)
47 {
48  if(ucPinName <= 7){ //TRISC
49 
50  if(bDirection){ //Input
51  REGISTER_TRISC |= (1<<ucPinName);
52  }
53  else{ //Output
54  REGISTER_TRISC &= ~(1<<ucPinName);
55  }
56  }
57  else if(ucPinName >= 8 && ucPinName <= 13){ //TRISB
58 
59  ucPinName -= 8;
60 
61  if(bDirection){ //Input
62  REGISTER_TRISB |= (1<<ucPinName);
63  }
64  else{ //Output
65  REGISTER_TRISB &= ~(1<<ucPinName);
66  }
67  }
68  else if(ucPinName >= 14 && ucPinName <= 20){ //TRISA, TRISE
69  ucPinName -= 14;
70  if(bDirection){ //Input
71  if(ucPinName <= 2){//RA0 to RA2
72  REGISTER_TRISA |= (1<<ucPinName);
73  }
74  else if(ucPinName == 3){ //RA5
75  REGISTER_TRISA |= (1<<(ucPinName+2));
76  }
77  else if(ucPinName >= 4 && ucPinName <= 6){ //RE0 to RE2
