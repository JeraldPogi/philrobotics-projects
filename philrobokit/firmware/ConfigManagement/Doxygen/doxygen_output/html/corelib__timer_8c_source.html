38 /* Timer Routine */
39  void timerInterruptHandler(void)
40  {
41  static int iMsCounter = 0;
42 
43  if(BIT_PIR1_TMR1IF&&BIT_PIE1_TMR1IE)
44  {
45  BIT_PIR1_TMR1IF = 0; // Clear Timer1 Interrupt Flag
46 
47  /* Increment US Timer */
48  uiTimerUs += CONST16_TIMER_INCREMENT;
49 
50  /* Increment MS Timer */
51  iMsCounter += CONST16_TIMER_INCREMENT;
52  if(iMsCounter >= 1000)
53  {
54  iMsCounter = 0;
55  uiTimerMs++;
56  }
57 
58  REGISTER_TMR1L = (CONST16_TIMER);
59  REGISTER_TMR1H = (CONST16_TIMER >> 8);
60  }
61  }
