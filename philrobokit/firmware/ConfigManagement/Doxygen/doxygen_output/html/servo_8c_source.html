69 void setupServo(enum servoModules ServoMod, unsigned char ServoPin, signed char DefaultAngle, unsigned char MinPulseWidth, unsigned char MaxPulseWidth, signed char MinAngle, signed char MaxAngle)
70 {
71  unsigned long TempBuffer;
72 
73  /* Initilize Parameters */
74  servo[ServoMod][SERVO_PIN] = ServoPin;
75 // servo[ServoMod][SERVO_MINPULSE] = MinPulseWidth;
76 // servo[ServoMod][SERVO_MAXPULSE] = MaxPulseWidth;
77 // servo[ServoMod][SERVO_MINANGLE] = MinAngle;
78 // servo[ServoMod][SERVO_MAXANGLE] = MaxAngle;
79 
80  /* Compute Servo Slope */
81  TempBuffer = (((signed long)MaxPulseWidth - MinPulseWidth)*128);
82  TempBuffer /= ((signed int)MaxAngle - MinAngle);
83  servo[ServoMod][SERVO_SLOPE] = (unsigned char)TempBuffer;
84 
85  /* Compute Servo Pulse Width */
86  servo[ServoMod][SERVO_PULSE] = (unsigned char)(((signed long)DefaultAngle*servo[ServoMod][SERVO_SLOPE]) / 128 + SERVO_PULSE_OFFSET);
87 
88  /* Initialize TMR2 for Servo Control */
89  if(0 == kickStarted)
90  {
91  setup8BitTimerDef(TIMER2,servoController);
92  setTimerValue(TIMER2,SERVO_MAX_PULSEWIDTH); // set pulse on period (kickstart)
93  kickStarted = 1;
94  }
95 }

117 /* Private Functions */
118 static void servoController()
119 {
120  /* Servo State Machine */
121  switch(servoState)
122  {
123  case SERVO_PULSEON:
124  {
125  setTimerValue(TIMER2,SERVO_MAX_PULSEWIDTH-servo[sequenceCounter][SERVO_PULSE]); // set pulse off period
126  setPinLow(servo[sequenceCounter][SERVO_PIN]);
127  servoState = SERVO_PULSEOFF;
128  break;
129  }
130  case SERVO_PULSEOFF:
131  {
132  /* Check Next Servo */
133  if(sequenceCounter < 8)
134  {
135  sequenceCounter++;
136  }
137  else
138  {
139  sequenceCounter = 0;
140  }
141 
142  /* Check if Angle is pulse < min */
143  if(servo[sequenceCounter][SERVO_PULSE] < SERVO_MIN_PULSEWIDTH)
144  {
145  /* skip servo module */
146  servo[sequenceCounter][SERVO_PULSE] = 0;
147  setTimerValue(TIMER2,SERVO_MAX_PULSEWIDTH); // low for 2.5mS
148  setPinLow(servo[sequenceCounter][SERVO_PIN]);
149  servoState = SERVO_PULSEOFF;
150  }
151  else
152  {
153  setTimerValue(TIMER2,servo[sequenceCounter][SERVO_PULSE]); // set pulse on period
154  setPinHigh(servo[sequenceCounter][SERVO_PIN]);
155  servoState = SERVO_PULSEON;
156  }
157  break;
158  }
159  default: /* must not be reached */
160  {
161  setTimerValue(TIMER2,SERVO_MAX_PULSEWIDTH); // low for 2.5mS
162  sequenceCounter = 8;
163  servoState = SERVO_PULSEOFF;
164  break;
165  }
166  }
167 }
