46 void setupADC(void)
47 {
48  REGISTER_TRISA = 0x3F;
49  REGISTER_TRISE = 0x07;
50 
51  BIT_ADCON0_ADON = 0; //AD Converter is temporarily disable
52 
53  BIT_ADCON1_PCFG0 = 0; //All Ports are analog, VDD as VREF
54  BIT_ADCON1_PCFG1 = 0;
55  BIT_ADCON1_PCFG2 = 0;
56  BIT_ADCON1_PCFG3 = 0;
57 
58  BIT_ADCON0_CHS0 = 0; //Channel 0(AN0) initially selected
59  BIT_ADCON0_CHS1 = 0;
60  BIT_ADCON0_CHS2 = 0;
61 
62  BIT_ADCON1_ADFM = 1; //Right Justified
63 
64  //if(FOSC_FREQ == 20000000) //if FOSC is @20MHz
65  {
66  BIT_ADCON0_ADCS0 = 0; //64Tosc @20MHz 11*TAD for 10bit, TAD = 20MHz/64
67  BIT_ADCON0_ADCS1 = 1;
68  BIT_ADCON1_ADCS2 = 1;
69  }
70 
71  BIT_ADCON0_ADON = 1; //AD Converter is enable
72 
73  //Enable Interrupt Here
74 
75  BIT_PIR1_ADIF = 0; //Clear ADC Interrupt Flag
76  BIT_PIE1_ADIE = 0; //Disable ADC Interrupt ByDefault
77 
78  BIT_INTCON_PEIE = 1; //Enable Peripheral Interrupt
79  BIT_INTCON_GIE = 1; //Enable Global Interrupt
80 }

91 void adcSetChannel(unsigned char ucChannel)
92 {
93  switch(ucChannel){
94  case 0x00: BIT_ADCON0_CHS0 = 0; BIT_ADCON0_CHS1 = 0; BIT_ADCON0_CHS2 = 0; break;
95  case 0x01: BIT_ADCON0_CHS0 = 1; BIT_ADCON0_CHS1 = 0; BIT_ADCON0_CHS2 = 0; break;
96  case 0x02: BIT_ADCON0_CHS0 = 0; BIT_ADCON0_CHS1 = 1; BIT_ADCON0_CHS2 = 0; break;
97  case 0x03: BIT_ADCON0_CHS0 = 0; BIT_ADCON0_CHS1 = 0; BIT_ADCON0_CHS2 = 1; break;
98  case 0x04: BIT_ADCON0_CHS0 = 1; BIT_ADCON0_CHS1 = 0; BIT_ADCON0_CHS2 = 1; break;
99  case 0x05: BIT_ADCON0_CHS0 = 0; BIT_ADCON0_CHS1 = 1; BIT_ADCON0_CHS2 = 1; break;
100  //case 0x06: BIT_ADCON0_CHS0 = 0; BIT_ADCON0_CHS1 = 1; BIT_ADCON0_CHS2 = 1; break;
101 
102  default: BIT_ADCON0_CHS0 = 0; BIT_ADCON0_CHS1 = 0; BIT_ADCON0_CHS2 = 0; break; //Default is AN0
103  }
104 }
